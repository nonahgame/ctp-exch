    <script>
        $(document).ready(function() {
            // Fetch trades
            $.getJSON('/trades', function(data) {
                if (data.error) {
                    $('#trades').html('<p>Error loading trades: ' + data.error + '</p>');
                    return;
                }
                let html = '<table><thead><tr>';
                const columns = ['time', 'action', 'symbol', 'price', 'profit', 'total_profit', 'message'];
                columns.forEach(col => {
                    html += '<th>' + col.replace('_', ' ').toUpperCase() + '</th>';
                });
                html += '</tr></thead><tbody>';
                data.forEach(trade => {
                    html += '<tr>';
                    columns.forEach(col => {
                        let value = trade[col];
                        if (['price', 'profit', 'total_profit'].includes(col) && value != null) {
                            value = Number(value).toFixed(2);
                        } else if (value == null) {
                            value = '';
                        }
                        html += '<td>' + value + '</td>';
                    });
                    html += '</tr>';
                });
                html += '</tbody></table>';
                $('#trades').html(html);
            }).fail(function() {
                $('#trades').html('<p>Failed to load trades.</p>');
            });

            // Fetch performance
            $.getJSON('/performance', function(data) {
                if (data.error) {
                    $('#performance').html('<p>Error loading performance: ' + data.error + '</p>');
                    return;
                }
                $('#performance').html('<pre>' + data.performance + '</pre>');
            }).fail(function() {
                $('#performance').html('<p>Failed to load performance data.</p>');
            });

            // Fetch status
            $.getJSON('/status', function(data) {
                if (data.error) {
                    $('#status').html('<p>Error loading status: ' + data.error + '</p>');
                    return;
                }
                let html = '<p><strong>Status:</strong> ' + data.status + '</p>';
                html += '<p><strong>Current Time:</strong> ' + data.current_time + '</p>';
                html += '<p><strong>Stop Time:</strong> ' + data.stop_time + '</p>';
                $('#status').html(html);
            }).fail(function() {
                $('#status').html('<p>Failed to load status.</p>');
            });
        });
    </script>
